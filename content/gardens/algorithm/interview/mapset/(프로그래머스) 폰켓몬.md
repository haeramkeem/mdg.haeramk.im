- 왜 이런 문제를 만드는지는 모르겠는데, 그냥 문제 읽고 생각나는 대로 풀면 된다.

```cpp
#include <vector>
#include <set>

using namespace std;

#define MIN(a, b) ((a) < (b) ? (a) : (b))

int solution(vector<int> nums)
{
	set<int> nums_set;

	for (auto num : nums) {
		nums_set.insert(num);
	}

	return MIN(nums_set.size(), nums.size() >> 1);
}
```
